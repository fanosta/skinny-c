43  (Skinny128CTR_t *ctr, const void *counter, unsigned size)
44 {
45  /* Validate the parameters */
46  if (!ctr || size > SKINNY128_BLOCK_SIZE)
47  return 0;
48 
49  /* Set the counter and reset the keystream to a block boundary */
50  if (counter) {
51  memset(ctr->counter, 0, SKINNY128_BLOCK_SIZE - size);
52  memcpy(ctr->counter + SKINNY128_BLOCK_SIZE - size, counter, size);
53  } else {
54  memset(ctr->counter, 0, SKINNY128_BLOCK_SIZE);
55  }
56  ctr->offset = SKINNY128_BLOCK_SIZE;
57  return 1;
58 }

60 int skinny128_ctr_encrypt
61  (void *output, const void *input, size_t size,
62  const Skinny128Key_t *ks, Skinny128CTR_t *ctr)
63 {
64  uint8_t *out = (uint8_t *)output;
65  const uint8_t *in = (const uint8_t *)input;
66 
67  /* Validate the parameters */
68  if (!output || !input || !ks || !ctr)
69  return 0;
70 
71  /* Encrypt the input in CTR mode to create the output */
72  while (size > 0) {
73  if (ctr->offset >= SKINNY128_BLOCK_SIZE) {
74  /* We need a new keystream block */
75  skinny128_ecb_encrypt(ctr->ecounter, ctr->counter, ks);
76  skinny128_inc_counter(ctr->counter, 1);
77 
78  /* XOR an entire keystream block in one go if possible */
79  if (size >= SKINNY128_BLOCK_SIZE) {
80  skinny128_xor(out, in, ctr->ecounter);
81  out += SKINNY128_BLOCK_SIZE;
82  in += SKINNY128_BLOCK_SIZE;
83  size -= SKINNY128_BLOCK_SIZE;
84  } else {
85  /* Last partial block in the request */
86  skinny_xor(out, in, ctr->ecounter, size);
87  ctr->offset = size;
88  break;
89  }
90  } else {
91  /* Left-over keystream data from the last request */
92  size_t temp = SKINNY128_BLOCK_SIZE - ctr->offset;
93  if (temp > size)
94  temp = size;
95  skinny_xor(out, in, ctr->ecounter + ctr->offset, temp);
96  ctr->offset += temp;
97  out += temp;
98  in += temp;
99  size -= temp;
100  }
101  }
102  return 1;
103 }
