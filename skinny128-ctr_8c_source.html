128  (Skinny128CTR_t *ctr, const void *counter, unsigned size)
129 {
130  Skinny128CTRCtx_t *ctx;
131 
132  /* Validate the parameters */
133  if (!ctr || size > SKINNY128_BLOCK_SIZE)
134  return 0;
135  ctx = ctr->ctx;
136  if (!ctx)
137  return 0;
138 
139  /* Set the counter and reset the keystream to a block boundary */
140  if (counter) {
141  memset(ctx->counter, 0, SKINNY128_BLOCK_SIZE - size);
142  memcpy(ctx->counter + SKINNY128_BLOCK_SIZE - size, counter, size);
143  } else {
144  memset(ctx->counter, 0, SKINNY128_BLOCK_SIZE);

150 int skinny128_ctr_encrypt
151  (void *output, const void *input, size_t size, Skinny128CTR_t *ctr)
152 {
153  Skinny128CTRCtx_t *ctx;
154  uint8_t *out = (uint8_t *)output;
155  const uint8_t *in = (const uint8_t *)input;
156 
157  /* Validate the parameters */
158  if (!output || !input || !ctr)
159  return 0;
160  ctx = ctr->ctx;
161  if (!ctx)
162  return 0;
163 
164  /* Encrypt the input in CTR mode to create the output */
165  while (size > 0) {
166  if (ctx->offset >= SKINNY128_BLOCK_SIZE) {
167  /* We need a new keystream block */
168  skinny128_ecb_encrypt(ctx->ecounter, ctx->counter, &(ctx->kt.ks));
169  skinny128_inc_counter(ctx->counter, 1);
170 
171  /* XOR an entire keystream block in one go if possible */
172  if (size >= SKINNY128_BLOCK_SIZE) {
173  skinny128_xor(out, in, ctx->ecounter);
174  out += SKINNY128_BLOCK_SIZE;
175  in += SKINNY128_BLOCK_SIZE;
176  size -= SKINNY128_BLOCK_SIZE;
177  } else {
178  /* Last partial block in the request */
179  skinny_xor(out, in, ctx->ecounter, size);
180  ctx->offset = size;
181  break;
182  }
183  } else {
184  /* Left-over keystream data from the last request */
185  size_t temp = SKINNY128_BLOCK_SIZE - ctx->offset;
186  if (temp > size)
187  temp = size;
188  skinny_xor(out, in, ctx->ecounter + ctx->offset, temp);
189  ctx->offset += temp;
190  out += temp;
191  in += temp;
192  size -= temp;
193  }
194  }
195  return 1;
196 }
