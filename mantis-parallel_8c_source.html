94 int mantis_parallel_ecb_crypt
95  (void *output, const void *input, const void *tweak, size_t size,
96  const MantisParallelECB_t *ecb)
97 {
98  const MantisKey_t *ks;
99  const MantisParallelECBVtable_t *vtable;
100 
101  /* Validate the parameters */
102  if (!ecb || !ecb->ctx || (size % MANTIS_BLOCK_SIZE) != 0)
103  return 0;
104  ks = ecb->ctx;
105 
106  /* Process major blocks with the vectorized back end */
107  vtable = ecb->vtable;
108  if (vtable) {
109  size_t psize = ecb->parallel_size;
110  while (size >= psize) {
111  (*(vtable->crypt))(output, input, tweak, ks);
112  output += psize;
113  input += psize;
114  tweak += psize;
115  size -= psize;
116  }
117  }
118 
119  /* Process any left-over blocks with the non-parallel implementation */
120  while (size >= MANTIS_BLOCK_SIZE) {
121  mantis_ecb_crypt_tweaked(output, input, tweak, ks);
122  output += MANTIS_BLOCK_SIZE;
123  input += MANTIS_BLOCK_SIZE;
124  tweak += MANTIS_BLOCK_SIZE;
125  size -= MANTIS_BLOCK_SIZE;
126  }
127  return 1;
128 }
